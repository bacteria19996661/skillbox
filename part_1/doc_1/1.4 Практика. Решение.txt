# 3.
#
# print('Hello World")

# Кавычки должны быть одинаковыми слева и справа от строки
print("Hello World")
# При этом не так важно какие именно мы выберем
print('Hello World')

# А вот если бы у нас внутри строки были кавычки - то этот выбор бы стал действительно важным:
# print('Hello' World')  # так код работать не будет
# print("Hello" World")  # и так тоже не будет

# Но если мы используем снаружи кавычки не такие как внутри строки, то всё будет ок:
print("Hello' World")
print('Hello" World')
# Теперь пайтон легко поймёт какие кавычки должны быть частью строки, а какие будут значит начало/конец строки
